123 template<typename T>
124 void AddComponent(EntityID id, const T& value) {
125 if (ComponentTypeRegistry::IsSparse<T>()) {
126 GetSparseStore<T>().Add(id, value);
127 return;
128 }
129
130 ComponentTypeID typeID = ComponentTypeRegistry::GetID<T>();
131 ComponentMask currentMask = GetMask(id);
132 currentMask.set(typeID);
133
134 Archetype* newArch = archetypeManager.GetOrCreate(currentMask);
135 ArchetypeChunk* newChunk = newArch->GetOrCreateChunk();
136
137 EntityLocation oldLoc = locations.at(id);
138 ArchetypeChunk* oldChunk = oldLoc.chunk;
139 size_t oldIndex = oldLoc.index;
140
141 size_t newIndex = newChunk->AddEntity(id);
142
143 // 必要なデータのみ移動
144 auto& oldChunkLayout = ArchetypeChunk::Accessor::GetLayoutInfo(oldChunk);
145 auto& newChunkLayout = ArchetypeChunk::Accessor::GetLayoutInfo(newChunk);
146 for (const auto& [comp, info] : oldChunkLayout) {
147 size_t index = 0;
148 for (const auto& i : info)
149 {
150 auto src = ArchetypeChunk::Accessor::GetBuffer(oldChunk) + i.offset;
151 auto dst = ArchetypeChunk::Accessor::GetBuffer(newChunk) + newChunkLayout.at(comp).get(index).offset;
152 std::memcpy(static_cast<uint8_t*>(dst) + newIndex * i.stride,
153 static_cast<const uint8_t*>(src) + oldIndex * i.stride,
154 i.stride);
155
156 index++;
157 }
158 }
159
160 newChunk->GetColumn<T>()[newIndex] = value;
161
162 oldChunk->RemoveEntitySwapPop(oldIndex);
163 locations[id] = { newChunk, newIndex };
164 }

171 template<typename T>
172 void RemoveComponent(EntityID id) {
173 ComponentTypeID typeID = ComponentTypeRegistry::GetID<T>();
174 ComponentMask oldMask = GetMask(id);
175 if (!oldMask.test(typeID)) return; // 持ってないなら何もしない
176
177 ComponentMask newMask = oldMask;
178 newMask.reset(typeID);
179
180 if (ComponentTypeRegistry::IsSparseMask(ComponentMask().set(typeID))) {
181 GetSparseStore<T>().Remove(id);
182 return;
183 }
184
185 EntityLocation oldLoc = locations.at(id);
186 ArchetypeChunk* oldChunk = oldLoc.chunk;
187 size_t oldIndex = oldLoc.index;
188
189 Archetype* newArch = archetypeManager.GetOrCreate(newMask);
190 ArchetypeChunk* newChunk = newArch->GetOrCreateChunk();
191 size_t newIndex = newChunk->AddEntity(id);
192
193 // 必要なデータのみ移動
194 const auto& oldChunkLayout = ArchetypeChunk::Accessor::GetLayoutInfo(oldChunk);
195 const auto& newChunkLayout = ArchetypeChunk::Accessor::GetLayoutInfo(newChunk);
196 for (const auto& [comp, info] : oldChunkLayout) {
197 if (comp == typeID) continue;
198
199 size_t index = 0;
200 for (const auto& i : info)
201 {
202 auto src = ArchetypeChunk::Accessor::GetBuffer(oldChunk) + i.offset;
203 auto dst = ArchetypeChunk::Accessor::GetBuffer(newChunk) + newChunkLayout.at(comp).get(index).offset;
204 std::memcpy(static_cast<uint8_t*>(dst) + newIndex * i.stride,
205 static_cast<const uint8_t*>(src) + oldIndex * i.stride,
206 i.stride);
207
208 index++;
209 }
210 }
211
212 oldChunk->RemoveEntitySwapPop(oldIndex);

281 template<typename Variant, typename U, size_t Index>
282 void ExpansionMemberImpl(std::vector<Variant>& vec, U& value) {
283 auto ptr = std::get<Index>(U::member_ptr_tuple);
284 static_assert(std::is_member_object_pointer_v<decltype(ptr)>);
285 auto& member = value.*ptr;
286
287 using MemberT = std::remove_cvref_t<decltype(member)>;
288
289 if constexpr (IsSoAComponent<MemberT>) {
290 ExpansionMember<Variant>(vec, member, std::make_index_sequence<std::tuple_size_v<typename MemberT::tuple_type>>{});
291 }
292 else if constexpr (IsPrimitive<MemberT>) {
293 vec.push_back(member);
294 }
295 else {
296 static_assert(std::is_same_v<MemberT, void>, "Unsupported member type: must be primitive or reflectable (HasMemberPtr)");
297 }
298 }

328 template<typename T, typename SoATuple, std::size_t... Is>
329 void MemorySetChunkSoA(T* base, size_t index, const T& value, size_t capacity, std::index_sequence<Is...>) {
330 auto charBase = reinterpret_cast<BufferType*>(base);
331 size_t offset = 0;
332 auto members = GetMember(value);
333 for (const auto& v : members) {
334 std::visit([&](auto&& val) {
335 using ValType = std::remove_cvref_t<decltype(val)>;
336 offset = AlignTo(offset, alignof(ValType));
337 *(reinterpret_cast<ValType*>(charBase + offset) + index) = val;
338 offset += sizeof(ValType) * capacity;
339 }, v);
340 }
